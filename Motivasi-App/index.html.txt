        // Dynamic Field Utility (Handles "Kustom" Select Fields)
        function toggleCustomField(selectElement, containerId) {
            const container = document.getElementById(containerId);
            if (!container) return;
            const input = container.querySelector('input');
            if (selectElement.value === 'custom') {
                container.classList.remove('hidden');
                input.setAttribute('required', 'true');
                input.focus();
            } else {
                container.classList.add('hidden');
                input.removeAttribute('required');
                input.value = '';
            }
        }

        // Helper to populate custom parameters dynamically
        function setSelectValueOrCustom(selectId, value, customInputId, customContainerId) {
            const selectEl = document.getElementById(selectId);
            const customInput = document.getElementById(customInputId);
            const customContainer = document.getElementById(customContainerId);
            
            if (!selectEl) return;
            
            let found = false;
            for (let i = 0; i < selectEl.options.length; i++) {
                if (selectEl.options[i].value === value) {
                    selectEl.value = value;
                    found = true;
                    break;
                }
            }
            
            if (!found && value) {
                selectEl.value = 'custom';
                if (customInput) customInput.value = value;
                if (customContainer) {
                    customContainer.classList.remove('hidden');
                    customInput.setAttribute('required', 'true');
                }
            } else {
                selectEl.value = value || selectEl.options[0].value;
                if (customInput) customInput.value = '';
                if (customContainer) {
                    customContainer.classList.add('hidden');
                    customInput.removeAttribute('required');
                }
            }
        }

        // WEB SPEECH INTELLIGENT MATCHING PROFILE
        function calculateVoiceProfile(gender, age) {
            let pitch = 1.0;
            let rate = 0.95;
            
            const g = gender.toLowerCase();
            const a = age.toLowerCase();
            
            // Adjust pitch according to Gender (Direct Actor Address)
            if (g.includes('wanita') || g.includes('hijab') || g.includes('perempuan') || g.includes('gadis')) {
                pitch = 1.25;
            } else if (g.includes('pria') || g.includes('laki') || g.includes('cowok') || g.includes('bapak')) {
                pitch = 0.82;
            } else {
                pitch = 1.0;
            }
            
            // Adjust speed & pitch multiplier based on Age group
            if (a.includes('anak') || g.includes('anak') || a.includes('10-15') || a.includes('muda') || a.includes('20-25')) {
                rate = 1.08;
                if (a.includes('anak') || g.includes('anak')) {
                    pitch += 0.2;
                }
            } else if (a.includes('paruh baya') || a.includes('tua') || a.includes('40+') || a.includes('60')) {
                rate = 0.78;
            }
            
            return { pitch, rate };
        }

        // DYNAMIC AUDIO SPEAKER GENERATOR
        function speakDirectly(text) {
            if ('speechSynthesis' in window) {
                window.speechSynthesis.cancel();

                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'id-ID';
                
                // Match the configured dynamic actor parameters computed on generation
                utterance.pitch = currentProjectVoiceParams.pitch;
                
                // Incorporate rate slider speed with character-based rate
                const speedEl = document.getElementById('voSpeed');
                const speedMultiplier = speedEl ? parseFloat(speedEl.value) : 0.95;
                utterance.rate = currentProjectVoiceParams.rate * (speedMultiplier / 0.95);

                const voices = window.speechSynthesis.getVoices();
                const idVoice = voices.find(v => v.lang.includes('id') || v.lang.includes('ID'));
                if (idVoice) utterance.voice = idVoice;

                window.speechSynthesis.speak(utterance);
                currentlyPlayingSpeech = utterance;
            } else {
                Swal.fire({
                    title: 'Web Speech API Tidak Didukung',
                    text: 'Peramban Anda tidak mendukung penyuaraan otomatis.',
                    icon: 'warning',
                    background: '#0A0A0A',
                    color: '#FFF',
                    confirmButtonColor: '#39FF14'
                });
            }
        }

        function playAllVoiceovers() {
            let fullCombined = "";
            const accordionElements = document.querySelectorAll('#sceneAccordionContainer .p-3');

        function syncWithGAS() {
            Swal.fire({
                title: 'Menghubungkan ke Cloud...',
                text: 'Membaca antrean dan mensinkronisasikan riwayat pengerjaan dengan Google Sheets Spreadsheet...',
                allowOutsideClick: false,
                background: '#0A0A0A',
                color: '#FFF',
                didOpen: () => {
                    Swal.showLoading();
                }
            });

            if (typeof google !== 'undefined' && google.script && google.script.run) {
                google.script.run
                    .withSuccessHandler(function(response) {
                        Swal.fire({
                            title: 'Sinkronisasi Berhasil!',
                            text: 'Seluruh riwayat lokal telah diselaraskan dengan Google Apps Script secara aman.',
                            icon: 'success',
                            background: '#0A0A0A',
                            color: '#FFF',
                            confirmButtonColor: '#39FF14'
                        });
                    })
                    .withFailureHandler(function(err) {
                        Swal.fire({
                            title: 'Kegagalan Sinkronisasi',
                            text: `Detail Error: ${err.message || err}. Data tetap tersimpan aman secara lokal.`,
                            icon: 'error',
                            background: '#0A0A0A',
                            color: '#FFF',
                            confirmButtonColor: '#39FF14'
                        });
                    })
                    .saveProjectsToSheets(JSON.stringify(dbHistory));
            } else {
                setTimeout(() => {
                    Swal.fire({
                        title: 'Tersinkronisasi Secara Lokal',
                        text: 'Cloud fallback aktif. Seluruh data disimpan rapi dalam LocalStorage. Hubungkan Google Apps Script untuk sinkronisasi Google Sheets.',
                        icon: 'info',
                        background: '#0A0A0A',
                        color: '#FFF',
                        confirmButtonColor: '#39FF14'
                    });
                }, 1500);
            }
        }
    </script>
